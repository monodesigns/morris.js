Bar charts with negative values

// Use Morris.Bar
Morris.Bar({
  element: 'graph',
  data: [
    {x: '2011 Q1', y: -3, z: 2},
    {x: '2011 Q2', y: 2, z: 0},
    {x: '2011 Q3', y: 0, z: 3},
    {x: '2011 Q4', y: 2, z: 4}
  ],
  stacked: true,
  xkey: 'x',
  ykeys: ['y', 'z'],
  labels: ['Y', 'Z']
}).on('click', function(i, row){
  console.log(i, row);
});
